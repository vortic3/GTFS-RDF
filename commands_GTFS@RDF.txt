FILE ( #stops , stops.txt)
FILE ( #stop_times , stop_times.txt)
FILE ( #trips , trips.txt)
FILE ( #routes , routes.txt)
FILE ( #agency , agency.txt)
PREFIX(gn,http://www.geonames.org/ontology#)
PREFIX(mao,http://www.vortic3.com/MANTO/)
PREFIX(sch,https://schema.org/)
PREFIX(wgs84_pos,http://www.w3.org/2003/01/geo/wgs84_pos#)
PREFIX(fnml,http://semweb.mmlab.be/ns/fnml#) 
PREFIX(fno,http://w3id.org/function/ontology#) 
PREFIX(grel,http://users.ugent.be/~bjdmeest/function/grel.ttl#) 
PREFIX(dbo,http://dbpedia.org/resource/)
SUBJECT(#stops.stop_id,mao:stopPlace)
PREDICATE-OBJECT(PREDICATE(sch:name),OBJECT(#stops.stop_name))
PREDICATE-OBJECT(PREDICATE(wgs84_pos:lat),OBJECT(#stops.stop_lat))
PREDICATE-OBJECT(PREDICATE(wgs84_pos:lon),OBJECT(#stops.stop_lon))
PREDICATE-OBJECT(PREDICATE(gn:timeZone),OBJECT(dbo:#stops.stop_timezone))
PREDICATE-OBJECT(PREDICATE(gn:parentFeature),OBJECT(mao:#stops.parent_station))
PREDICATE-OBJECT(PREDICATE(mao:stopPlaceCode),OBJECT(#stops.stop_code))
PREDICATE-OBJECT(PREDICATE(mao:address),OBJECT(#stops.stop_desc))
PREDICATE-OBJECT(PREDICATE(mao:tariffZone),OBJECT(#stops.zone_id))
PREDICATE-OBJECT(PREDICATE(mao:infoLink),OBJECT(#stops.stop_url))
PREDICATE-OBJECT(PREDICATE(mao:locationType),OBJECT(#stops.location_type,VALUES(,0,1,2),ENUMS(quay,quay,stopPlace,entrance)))
PREDICATE-OBJECT(PREDICATE(mao:ofLine),OBJECT(QUERY(MATCH(#stops.stop_id,#stop_times.stop_id),MATCH(#stop_times.trip_id,#trips.trip_id),MATCH(#trips.route_id,#routes.route_id),mao:#routes.route_id)))
PREDICATE-OBJECT(PREDICATE(mao:transportFor),OBJECT(QUERY(MATCH(#stops.stop_id,#stop_times.stop_id),MATCH(#stop_times.trip_id,#trips.trip_id),MATCH(#trips.route_id,#routes.route_id),#routes.route_type,VALUES(0,1,2,3,4,5,6,7),ENUMS(lightRail,Metro,Rail,Bus,Ferry,cableTram,aerialLift,Funicular))))
PREDICATE-OBJECT(PREDICATE(mao:connectionLink),OBJECT(QUERY(MATCH(#stops.stop_id,#stops.parent_station),MATCH(#stops.stop_id,#stop_times.stop_id),MATCH(#stop_times.trip_id,#trips.trip_id),MATCH(#trips.route_id,#routes.route_id),CONTAINER(BAG,#routes.route_short_name))))
PREDICATE-OBJECT(PREDICATE(mao:transfer),OBJECT(QUERY(MATCH(#stops.stop_id,#stops.parent_station),MATCH(#stops.stop_id,#stop_times.stop_id),MATCH(#stop_times.trip_id,#trips.trip_id),MATCH(#trips.route_id,#routes.route_id),CONTAINER(BAG,#routes.route_type),VALUES(0,1,2,3,4,5,6,7),ENUMS(lightRail,Metro,Rail,Bus,Ferry,cableTram,aerialLift,Funicular))))
PREDICATE-OBJECT(PREDICATE(mao:wheelchairAccess),OBJECT(#stops.wheelchair_boarding,VALUES(,1,2),ENUMS(,true,false)))
SUBJECT(#routes.route_id,mao:line)
PREDICATE-OBJECT(PREDICATE(sch:name),OBJECT(#routes.route_short_name))
PREDICATE-OBJECT(PREDICATE(sch:description),OBJECT(#routes.route_long_name))
PREDICATE-OBJECT(PREDICATE(mao:authority),OBJECT(QUERY(MATCH(#routes.agency_id,#agency.agency_id),#agency.agency_name)))
PREDICATE-OBJECT(PREDICATE(gn:timeZone),OBJECT(QUERY(MATCH(#routes.agency_id,#agency.agency_id),dbo:#agency.agency_timezone)))
PREDICATE-OBJECT(PREDICATE(mao:infoLink),OBJECT(#routes.route_url))
PREDICATE-OBJECT(PREDICATE(mao:transportFor),OBJECT(#routes.route_type,VALUES(0,1,2,3,4,5,6,7),ENUMS(lightRail,Metro,Rail,Bus,Ferry,cableTram,aerialLift,Funicular)))
PREDICATE-OBJECT(PREDICATE(mao:infoLinkAgency),OBJECT(QUERY(MATCH(#routes.agency_id,#agency.agency_id),#agency.agency_url)))
PREDICATE-OBJECT(PREDICATE(mao:sequence),OBJECT(CONTAINER(SEQ,mao:#routes.route_id)))
